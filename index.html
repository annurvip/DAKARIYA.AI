const express = require("express"); const bcrypt = require("bcryptjs"); const jwt = require("jsonwebtoken"); const app = express(); app.use(express.json()); /* ====================== IN-MEMORY DATABASE ====================== */ let users = []; let events = []; /* ====================== FRONTEND UI (SINGLE FILE) ====================== */ const html = ` DAKARIYA AI

DAKARIYA AI SOC

Register

Login

Simulate Attack Event

Live SOC Feed

`; /* ====================== ROUTES ====================== */ app.get("/", (req,res)=>{ res.send(html); }); /* REGISTER */ app.post("/register", async (req,res)=>{ const hashed = await bcrypt.hash(req.body.password || "1234",10); users.push({ id: users.length+1, email: req.body.email, password: hashed }); res.json({msg:"user created"}); }); /* LOGIN */ app.post("/login", async (req,res)=>{ const user = users.find(u=>u.email===req.body.email); if(!user) return res.json({msg:"not found"}); const ok = await bcrypt.compare(req.body.password,user.password); if(!ok) return res.json({msg:"wrong password"}); const token = jwt.sign({id:user.id},"SECRET"); res.json({token,user}); }); /* RISK ENGINE */ function risk(){ return Math.floor(Math.random()*100); } /* EVENT */ app.post("/event",(req,res)=>{ events.unshift({ risk: risk(), time: new Date() }); if(events.length>20) events.pop(); res.json({ok:true}); }); /* EVENTS */ app.get("/events",(req,res)=>{ res.json(events); }); /* START */ app.listen(5000,()=>{ console.log("DAKARIYA AI RUNNING 🚀"); });
